// controllers/PedidosController.ts
import { Request, Reply } from 'fastify';
import Pedido from '../models/Pedido';

class PedidosController {
  async index(request: Request, reply: Reply) {
    const pedidos = await Pedido.find();

    return reply.send(pedidos);
  }

  async show(request: Request, reply: Reply) {
    const { id } = request.params;

    const pedido = await Pedido.findById(id);

    return reply.send(pedido);
  }

  async create(request: Request, reply: Reply) {
    const pedido = await Pedido.create(request.body);

    return reply.status(201).send(pedido);
  }

  async update(request: Request, reply: Reply) {
    const { id } = request.params;

    const pedido = await Pedido.findByIdAndUpdate(id, request.body, {
      new: true,
    });

    return reply.send(pedido);
  }

  async delete(request: Request, reply: Reply) {
    const { id } = request.params;

    await Pedido.findByIdAndDelete(id);

    return reply.status(204).send();
  }
}

export default PedidosController;

// controllers/AutenticacaoController.ts
import { Request, Reply } from 'fastify';
import bcrypt from 'bcryptjs';
import jwt from 'jsonwebtoken';
import Usuario from '../models/Usuario';

class AutenticacaoController {
  async cadastro(request: Request, reply: Reply) {
    const { email, senha } = request.body;

    const usuario = await Usuario.findOne({ email });

    if (usuario) {
      return reply.status(400).send({ mensagem: 'Usuário já existe' });
    }

    const salt = await bcrypt.genSalt(10);
    const hash = await bcrypt.hash(senha, salt);

    const novoUsuario = await Usuario.create({ email, senha: hash });

    return reply.status(201).send(novoUsuario);
  }

  async login(request: Request, reply: Reply) {
    const { email, senha } = request.body;

    const usuario = await Usuario.findOne({ email });

    if (!usuario) {
      return reply.status(400).send({ mensagem: 'Usuário não encontrado' });
    }

    const senhaValida = await bcrypt.compare(senha, usuario.senha);

    if (!senhaValida) {
      return reply.status(400).send({ mensagem: 'Senha inválida' });
    }

    const token = jwt.sign({ id: usuario._id }, process.env.JWT_SECRET, {
      expiresIn: '1h',
    });

    return reply.send({ token });
  }
}

export default AutenticacaoController;
